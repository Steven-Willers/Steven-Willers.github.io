import matplotlib.pyplot as plt
import numpy as np

# Generate random training loss data
training_loss = np.random.rand(100) * 100

# Check for NaN or infinity values
if np.isnan(training_loss).any():
    nan_indices = np.where(np.isnan(training_loss))[0]
    for i in nan_indices:
        training_loss[i] = 0  # Replace NaN with 0

if np.isinf(training_loss).any():
    inf_indices = np.where(np.isinf(training_loss))[0]
    for i in inf_indices:
        training_loss[i] = np.nanmedian(training_loss)  # Replace inf with median value

# Calculate the moving average (optional)
window_size = 10
smoothed_loss = np.convolve(training_loss, np.ones(window_size) / window_size, mode='valid')


plt.plot(training_loss, label="Raw Loss")
if len(smoothed_loss) > 0:
    plt.plot(smoothed_loss, label="Smoothed Loss (window size: {})".format(window_size))


if np.isnan(training_loss).any():
    nan_indices = np.where(np.isnan(training_loss))[0]
    for i in nan_indices:
        plt.axvline(x=i, color='red', linestyle='--', label='NaN Value')

if np.isinf(training_loss).any():
    inf_indices = np.where(np.isinf(training_loss))[0]
    for i in inf_indices:
        plt.axvline(x=i, color='orange', linestyle='--', label='Infinity Value')

plt.xlabel("Epoch")
plt.ylabel("Loss")
plt.title("Training Loss Curve")
plt.legend()
plt.grid(True)


if any(training_loss[-window_size:] == training_loss[-window_size - 1]):
    plt.text(len(training_loss) - window_size // 2, training_loss[-window_size], "Potential Plateau", ha='center', va='bottom', fontsize=10)

if np.diff(training_loss[-window_size:])[-1] > 10 * np.mean(np.diff(training_loss)):  # Adjust threshold as needed
    plt.text(len(training_loss) - window_size // 2, training_loss[-window_size], "Possible Divergence", ha='center', va='top', fontsize=10)
